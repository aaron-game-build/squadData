# 一、原始update数据样例与初步注释


```json
{
  "values": [								 // 待解析数据信息
    [										// 全局的玩家单位信息
      0, 										// 玩家id
      1, 										// 角色id（不是全局唯一）
      4, 										// （位置信息）行数
      2, 										// （位置信息）列数
      400, 										// 攻击力
      0, 										// 防御力
      800, 										// 当前血量
      2, 										// 移动力
      800, 										// 购买价格
      103										// 单位类型
    ],
    [0, 0, 7, 0, 0, 0, 2000, 0, 0, 100],		// 全局的玩家单位信息
    [1, 1, 0, 6, 500, 0, 400, 1, 800, 101],		// 全局的玩家单位信息
    [1, 0, 0, 7, 0, 0, 2000, 0, 0, 100],		// 全局的玩家单位信息
    [1, 3, 4, 5, 400, 0, 800, 2, 800, 103],		// 全局的玩家单位信息
    [1, 0],									 // 当前可行动的单位列表，使用单位的 角色ID
    [0], [0],								 // 玩家1 和 玩家2 的基地是否被攻破
    [0], [0],								 // 玩家1 和 玩家2 的剩余金币数量
    [9],									// 游戏全局的 round 数
    [4],									// 当前玩家的 turn 数
    [1, -1, 0, 0, 0]						 // 上一个 round 的用户操作数据
  ],
  "attributes": [							 // 上面数据的切割方式
    [
        5, 									// 当前环境中有5个单位
        10									// 每个单位有10个属性
    ],
    [2],									// 当前可行动的单位数
    [2],									// 双方基地被攻破的情况
    [2],									// 双方的剩余金币数量
    [2],									// round 和 turn
    [5]										// 上一个用户的操作数据
  ]
}
```
